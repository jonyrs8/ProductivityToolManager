-- Criando a tabela DEPARTMENT
CREATE TABLE DEPARTMENT (
    DEPARTMENT_NAME VARCHAR(255) PRIMARY KEY
);

-- Criando a tabela TASKCODE
CREATE TABLE TASKCODE (
    TASK_CODE VARCHAR(50) PRIMARY KEY
);

-- Criando a tabela WORKER
CREATE TABLE WORKER (
    WORKER_ID VARCHAR (10) PRIMARY KEY,
    WORKER_NAME VARCHAR(255)
);

-- Criando a tabela TASKS com chaves estrangeiras
CREATE TABLE TASKS (
    TASK_ID INT PRIMARY KEY,
    DEPARTMENT_NAME VARCHAR(255),
    TASK_CODE VARCHAR(50),
    WORKER_ID VARCHAR(10),
    START_TIME DATETIME,
    END_TIME DATETIME,
    FOREIGN KEY (DEPARTMENT_NAME) REFERENCES DEPARTMENT(DEPARTMENT_NAME),
    FOREIGN KEY (TASK_CODE) REFERENCES TASKCODE(TASK_CODE),
    FOREIGN KEY (WORKER_ID) REFERENCES WORKER(WORKER_ID)
);


COMMIT;
